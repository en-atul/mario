import { Meta, Title } from "@storybook/addon-docs/blocks";
import { fontSize, fontWeight } from "./tokens/typography";
import React from "react";

<Meta title="Tokens/Typography" />

<Title>Typography</Title>

Typography tokens define font sizes, line heights, and font weights used throughout the design system.

## Font Sizes

<div className="grid grid-cols-[6rem_1fr] gap-x-4 gap-y-6">
  {Object.entries(fontSize).map(([name, [size, { lineHeight }]]) => (
    <React.Fragment key={name}>
      {/* Token name */}
      <div className="text-sm font-medium font-mono">{name}</div>

      {/* Meta */}
      <div className="text-xs text-slate-600 font-mono">
        {size} / {lineHeight}
      </div>

      {/* Sample (spans both columns) */}
      <div
        className="col-span-2 text-slate-900 border-b border-slate-200 pb-4"
        style={{ fontSize: size, lineHeight }}
      >
        The quick brown fox jumps over the lazy dog
      </div>
    </React.Fragment>
  ))}
</div>


## Font Weights

<div className="grid grid-cols-[8rem_6rem_1fr] gap-x-4 gap-y-3 items-center">
  {Object.entries(fontWeight).map(([name, value]) => (
    <React.Fragment key={name}>
      {/* Token name */}
      <div className="text-sm font-medium font-mono">{name}</div>

      {/* Value */}
      <div className="text-sm text-slate-600 font-mono">{value}</div>

      {/* Sample */}
      <div
        className="text-lg text-slate-900"
        style={{ fontWeight: value }}
      >
        The quick brown fox jumps over the lazy dog
      </div>
    </React.Fragment>
  ))}
</div>
