import { Meta, Title } from "@storybook/addon-docs/blocks";
import { spacing } from "./tokens/spacing";
import React from "react";

<Meta title="Tokens/Spacing" />

<Title>Spacing Scale</Title>

Spacing tokens define consistent spacing values used throughout the design system.

<div className="grid grid-cols-[4rem_8rem_1fr] gap-x-4 gap-y-3 items-center">
  {Object.entries(spacing)
    .sort(([a], [b]) => {
      const numA = parseInt(a) || 0;
      const numB = parseInt(b) || 0;
      return numA - numB;
    })
    .map(([name, value]) => (
      <React.Fragment key={name}>
        <div className="text-sm font-medium font-mono">{name}</div>
        <div className="text-sm text-slate-600 font-mono">{value}</div>
        <div className="flex items-center">
          <div
            className="h-4 bg-primary"
            style={{ width: value }}
          />
        </div>
      </React.Fragment>
    ))}
</div>
